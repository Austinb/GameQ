 77: 
 78:         // Set the current index
 79:         $index_current = 9;
 80: 
 81:         // Pull the team count
 82:         $teamCount = $result->get('num_teams');
 83: 
 84:         // Loop for the number of teams found, increment along the way
 85:         for ($id = 1; $id <= $teamCount; $id++, $index_current++) {
 86:             // Shows the tickets
 87:             $result->addTeam('tickets', $items[$index_current]);

 89:             $result->addTeam('id', $id);

 91:

 93:         $result->add('targetscore', (int) $items[$index_current]);
 94:         $result->add('online', 1); // Forced true, it seems $words[$index_current + 1] is always empty
 95:         $result->add('ranked', (int) $items[$index_current + 2]);
 96:         $result->add('punkbuster', (int) $items[$index_current + 3]);
 97:         $result->add('password', (int) $items[$index_current + 4]);
 98:         $result->add('uptime', (int) $items[$index_current + 5]);
 99:         $result->add('roundtime', (int) $items[$index_current + 6]);

101:         $result->add('punkbuster_version', $items[$index_current + 8]);
102:         $result->add('join_queue', (int) $items[$index_current + 9]);
103:         $result->add('region', $items[$index_current + 10]);
