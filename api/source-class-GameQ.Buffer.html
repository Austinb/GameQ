432:      *
433:      * @return int

436:     public function readInt64()
437:     {
438: 
439:         // We have the pack 64-bit codes available. See: http://php.net/manual/en/function.pack.php
440:         if (version_compare(PHP_VERSION, '5.6.3') >= 0) {
441:             // Change the integer type we are looking up
442:             switch ($this->number_type) {
443:                 case self::NUMBER_TYPE_BIGENDIAN:
444:                     $type = 'Jint';
445:                     break;

447:                 case self::NUMBER_TYPE_LITTLEENDIAN:
448:                     $type = 'Pint';
449:                     break;
450: 
451:                 default:
452:                     $type = 'Qint';
453:             }
454: 
455:             $int64 = unpack($type, $this->read(8));
456: 
457:             $int = $int64['int'];
458: 
459:             unset($int64);

462:                 $high = $this->readInt32();
